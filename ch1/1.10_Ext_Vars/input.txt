#include <stdio.h>
// "hello world"
// This is a single-line comment
' '
''

int main() {

    int a = 5; // initialize a
    int b = 10; /* initialize b */

    printf("Hello world\n"); // print greeting

    printf("// this is not a comment\n");
    printf("/* this is not a comment either */\n");

    char c1 = '/';
    char c2 = '*';
    char c3 = '\'';
    char c4 = '"';

    printf("Escaped quote: \" // still not a comment\n");
    printf("Escaped backslash: \\ /* not comment */\n");

    /* This is a
       multi-line comment
       that spans several lines */

    int x = 100;

    /*
    Another comment block
    with // inside it
    and even /* fake nested */ text
    */

    printf("Mixing // and /* inside string */ still safe\n");

    int y = 20; /* comment */ int z = 30;

    ////// multiple slashes comment

    printf("End of program\n");

    return 0;
}
